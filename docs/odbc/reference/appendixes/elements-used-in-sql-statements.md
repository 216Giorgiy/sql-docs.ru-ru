# <a name="elements-used-in-sql-statements"></a>Элементы, используемые в инструкциях SQL
Следующие элементы используются в инструкциях SQL, перечисленные ранее.  
  
## <a name="element"></a>Элемент  
 *идентификатор для таблицы базы* :: = *определенное имя пользователя*  
  
 *Имя таблицы базы* :: = *base идентификатор таблицы*  
  
 *Логическое значение коэффициента* :: = [NOT] *основной логическое значение*  
  
 *Логическое значение первичный* :: = сравнения *-предикат* &#124; ( *условие поиска* )  
  
 *Логическое значение термин* :: = *логическое значение коэффициента* [AND *термин логическое значение*]  
  
 *символ строковый литерал* :: = '' {*символ*}... " (*символ* — это любой символ в кодировке драйвера или данных источника. Чтобы включать символ одинарной кавычки литерала ("") в символьном литерале строки, используйте два символа кавычки литерала ['' '].)  
  
 *Идентификатор столбца* :: = *определенное имя пользователя*  
  
 *Имя столбца* :: = [*имя таблицы*.] *идентификатор столбца*  
  
 *оператор сравнения* :: = < &#124; > &#124; \<= &#124; > = &#124; = &#124; <>  
  
 *Предикат сравнения* :: = *выражение* выражения оператора сравнения  
  
 *Тип данных* :: = *тип символьной строки* (*тип символьной строки* является любым типом данных, для которого столбец ««тип данных»» в результирующем наборе, возвращенные SQLGetTypeInfo является либо SQL_CHAR или SQL_VARCHAR.)  
  
 *цифра* :: = 0 &#124; 1 &#124; 2 &#124; 3 &#124; 4 &#124; 5 &#124; 6 &#124; 7 &#124; 8 &#124; 9  
  
 *динамический параметр* :: =?  
  
 *выражение* :: = термин &#124; выражения {+&#124;—} термин  
  
 *Коэффициент* :: = [*+*&#124;*—*]*основной*  
  
 *Вставить значение* :: =  
  
 *динамический параметр*  
  
 &#124;*литерала*  
  
 &AMP;#124;ЗНАЧЕНИЕ NULL  
  
 &AMP;#124;ПОЛЬЗОВАТЕЛЬ  
  
 *буква* :: = *буквы нижнем case &#124; верхний регистр букв*  
  
 *литерал* :: = *символ строковый литерал*  
  
 *буква нижнем case* :: = &#124; b &#124; c &#124; d &#124; e &#124; f &#124; g &#124; h &#124; я &#124; j &#124; k &#124; l &#124; m &#124; n &#124; o &#124; p &#124; q &#124; r &#124; s &#124; t &#124; u &#124; v &#124; w &#124; x &#124; y &#124; z  
  
 *предложение ORDER by* :: = ORDER BY *спецификации сортировки* [, *спецификации сортировки*]...  
  
 *основной* :: = *имя столбца*  
  
 &#124;*динамического параметра*  
  
 &#124;*литерала*  
  
 &#124;( *выражение* )  
  
 *условие поиска* :: = *boolean термин* [или *условие поиска*]  
  
 *список выбора* :: = \* &#124; *подчиненный список select* [, *подчиненный список select*]...  (*список выбора* не может содержать параметры.)  
  
 *подчиненный список SELECT* :: = *выражение*  
  
 *спецификации сортировки* :: = {*unsigned integer &#124; имя столбца*} [*ASC &#124; DESC*]  
  
 *Идентификатор таблицы* :: = *определенное имя пользователя*  
  
 *Имя таблицы* :: = *идентификатор таблицы*  
  
 *ссылка на таблицу* :: = *имя таблицы*  
  
 *Таблица список ссылок* :: = *ссылка на таблицу* [,*ссылка на таблицу*]...  
  
 *Термин* :: = *коэффициент* &#124; *термин* {\*&#124;*/*} *коэффициент*  
  
 *unsigned integer* :: = {*цифра*}  
  
 *верхний регистр букв* :: = *A &#124; B &#124; C &#124; D &#124; E &#124; F &#124; G &#124; H &#124; я &#124; J &#124; K &#124; L &#124; M &#124; N &#124; O &#124; P &#124;Q &#124; R &#124; S &#124; T &#124; U &#124; V &#124; W &#124; X &#124; Y &#124; Z*  
  
 *определенное имя пользователя* :: = *буквы*[*цифра* &#124; *буквы* &#124; *_*]...
